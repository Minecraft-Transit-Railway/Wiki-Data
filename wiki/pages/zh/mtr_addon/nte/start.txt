public class CommandAddTrain extends CommandBase {

    @Override
    public String getName() {
        return "addtrain"; // Komut adı
    }

    @Override
    public String getUsage(ICommandSender sender) {
        return "/addtrain <train_name>";
    }

    @Override
    public void execute(MinecraftServer server, ICommandSender sender, String[] args) throws CommandException {
        if (args.length < 1) {
            throw new CommandException("U-bahnj.");
        }

        String trainName = args[0];
        
        // Tren ekleme işlemi (mod API'sine göre)
        boolean success = NemoTransitAPI.addTrain(trainName); // Örnek API çağrısı

        if (success) {
            sender.sendMessage(new TextComponentString("Tren başarıyla eklendi: " + trainName));
        } else {
            sender.sendMessage(new TextComponentString("Tren eklenemedi: " + trainName));
        }
    }
}
